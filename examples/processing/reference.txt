//Hieronder een paar basisfuncties uit Processing om je op gang te helpen
    
    //Kleuren in Processing:
      background(color);     // stel de achtergrondkleur in
      fill(color);           // stel de kleur van de fill in
      noFill();              // zet de fill uit
      stroke(color);         // stel de kleur van de stroke in
      strokeWeight(weight);  // stel de dikte van de stroke in
      noStroke();            // zet de stroke uit
    
    //Kleuren in Processing:
      background(#3fc45b);                   // hexadecimaal
      background(#3fc45b, alpha);            // hexadecimaal + alpha(0-255)
      background(gray);                      // grijsschaalwaarde(0-255)
      background(gray, alpha);               // grijsschaalwaarde(0-255) + alpha(0-255)
      background(red, green, blue);          // rood(0-255) + groen(0-255) + blauw(0-255)
      background(red, green, blue, alpha);   // rood(0-255) + groen(0-255) + blauw(0-255) + alpha(0-255)
      
    //Vormen in Processing:
      line(x1, y1, x2, y2);          // tekent een lijn van punt (x1, y1) naar punt (x2, y2); LET OP: zonder stroke is de lijn niet zichtbaar
      rect(x, y, width, height);     // tekent een rechthoek op (x, y) met een breedte 'width' en hoogte 'height'
      ellipse(x, y, width, height);  // tekent een ellips op (x y) met een breedte 'width' en hoogte 'height'; tip: gebruik ellipse() met een lage width en height om punten te tekenen
      
    //Tekenmodus in Processing:  
      rectMode(CORNER);     // de linkerbovenhoek van de rechthoek wordt op de voorgeschreven positie (x, y) gezet; als je geen rectMode() opgeeft, wordt van rectMode(CORNER) uitgegaan
      rectMode(CENTER);     // het midden van de rechthoek wordt op de voorgeschreven positie (x, y) gezet
      ellipseMode(CORNER);  // de linkerbovenhoek van de ellips wordt op de voorgeschreven positie (x, y) gezet
      ellipseMode(CENTER);  // het midden van de ellips wordt op de voorgeschreven positie (x, y) gezet; als je geen ellipseMode() opgeeft, wordt van ellipsMode(CENTER) uitgegaan
